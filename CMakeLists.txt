PROJECT(QtHardMon)
cmake_minimum_required(VERSION 2.8)

FIND_PACKAGE(Qt4 REQUIRED)

SET(QtHardMon_SOURCES main.cpp QtHardMon.cpp)
SET(QtHardMon_HEADERS QtHardMon.h)
SET(QtHardMon_FORMS QtHardMonForm.ui)
SET(QtHardMon_RESOURCES QtHardMon.qrc)

QT4_WRAP_CPP(QtHardMon_HEADERS_MOC ${QtHardMon_HEADERS})
QT4_WRAP_UI(QtHardMon_FORMS_HEADERS ${QtHardMon_FORMS})
QT4_ADD_RESOURCES(QtHardMon_RESOURCES_RCC ${QtHardMon_RESOURCES})

INCLUDE(${QT_USE_FILE})
ADD_DEFINITIONS(${QT_DEFINITIONS})

ADD_EXECUTABLE(QtHardMon ${QtHardMon_SOURCES} 
  ${QtHardMon_HEADERS_MOC}
  ${QtHardMon_FORMS_HEADERS} 
  ${QtHardMon_RESOURCES_RCC})
TARGET_LINK_LIBRARIES(QtHardMon ${QT_LIBRARIES})

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})
